def numconv(df):  # Definindo funo que transforma todos os elementos do dataframe em uma nica lista.
    lista = df.values.tolist()  # Convertendo o datrafame em uma lista de listas com 4 elementos.
    lista = list(itertools.chain(*lista))  # Mesclando as listas internas e retornando somente uma lista grande.
    df = DataFrame (lista, columns=['numbers'])  # Definindo a lista como um dataframe de coluna chamada 'numbers'.
    df = df['numbers'].value_counts()  # Fazendo um novo dataframe com a contagem de repetio dos numeros do dataframe anterior.
    return df

def filtro(df,a,b,c,d,e):  # Definindo uma funo que filtra o perodo escolhido a partir dos anos contidos na coluna 'data'.
    searchfor = [str(a), str(b), str(c), str(d), str(e)]  # criando uma varivel de variveis, com as entradas do comando convertidas em strings
    df = df.reset_index()  # Convertendo a 'data' de ndice pra uma coluna normal.
    df = df[df.data.str.contains('|'.join(searchfor))]  # Buscando as linhas onde a data contm alguma das variveis de entrada informadas e definindo um novo dataframe.
    df = df.set_index('data')  # Retornando a coluna data como um indice das linhas.
    return df

def contador(listas):  # Definindo a funo que vai contar quantos valores do sorteio batem com os 6 valores mais sorteados do perodo 2015-2020.
    dic = [10, 37, 11, 35, 34, 6]  # Criando uma lista com os 6 valores mais recorrentes.
    count = 0  # Criando um contador
    contagem = []  # Criando uma lista com as contagens.
    for i in listas:  # Para cada row do dataframe:
        for h in i:   # Para cada elemento da row:
            if h in dic:  # Se o elemento estiver dentro da lista 'dic', onde inserimos os nmeros que mais so sorteados:
                count += 1  # Aumenta uma unidade no contador temporrio.
        contagem.append(count)  # Ao termino do loop for que confere cada elemento, adiciona  lista 'contagem', o valor acumulado no 'counts'.
        count = 0  # Reseta a varivel 'counts' para comear a contagem dos elementos da prxima row partindo de zero.
    return contagem

def sequencia(listas):  # Definindo a funo que vai inserir uma nova coluna no dataframe com os elementos que coincidem com os 6 mais recorrentes do perodo 2015-2020.
    dic = [10, 37, 11, 35, 34, 6]  # Criando uma lista com os 6 valores mais recorrentes.
    seq = []  # Criando uma lista que vai formar a coluna do dataframe.
    seq2 = []  # Criando uma segunda lista que vai conter cada nmero coincidente com cada valor.
    for i in listas:  # Para cada row do dataframe:
        for h in i:  # Para cada elemento da row:
            if h in dic:  # Se o elemento estiver dentro da lista 'dic', onde inserimos os nmeros que mais so sorteados: 
                seq2.append(h)  # Adicionar o elemento na lista 'seq2' temporria.
        seq.append(seq2)  # Adiciona a lista 'seq2' na lista 'seq'.
        seq2 = []  # Reseta a lista 'seq2' para uma nova analise da prxima row.
    return seq

3 - Cálculo de chance para cada tipo de acerto¶

Combinações possíveis com 6 dígitos da Mega Sena:¶

(60!)/(6!(60-6)!) = 50.063.860

Chance de acertar todos os 6 dígitos:¶

1/50063860 ou 0,000002%

Combinações possíveis com 5 digitos:¶

(60!)/(5!(60-5)!) = 5.461.512

Chance de acertar todos os 5 dígitos:¶

1/5461512 ou 0.00002%

Combinações possíveis com 4 digitos:¶

(60!)/(4!(60-4)!) = 487.635

Chance de acertar todos os 4 dígitos:¶

1/487635 ou 0.0002%

Conclusão até aqui:¶

Podemos ver que, por mais que alguns nmeros acabem se repetindo algumas vezes a mais, existem tantas combinaes que, a chance de acertar uma sequncia de 4 ou mais digitos tornasse, em sua maior parte, pura sorte.
